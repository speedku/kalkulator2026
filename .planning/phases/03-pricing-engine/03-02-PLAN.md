```tsx
"use client";
import { useRef } from "react";
import { create } from "zustand";
import { toast } from "sonner";
import { batchUpsertMarginsAction } from "@/lib/actions/price-lists";
import type { MarginMatrixEntry } from "@/types/price-lists";

type MatrixStore = {
  edits: Record<string, number>; // key: `${productGroupId}`, value: marginPercent
  isDirty: boolean;
  setMargin: (groupId: number, value: number) => void;
  reset: () => void;
};

function createMatrixStore(initial: Record<string, number>) {
  return create<MatrixStore>((set) => ({
    edits: initial,
    isDirty: false,
    setMargin: (groupId, value) =>
      set((s) => ({ edits: { ...s.edits, [String(groupId)]: value }, isDirty: true })),
    reset: () => set({ isDirty: false }),
  }));
}

// Props: priceListId, productGroups (all groups), existingMargins (current DB values)
// Build initial edits record from existingMargins
// Grid: <table> with one row per product group, columns: group name, margin % input, computed example multiplier
// Each <input type="number" min="0" max="99.99" step="0.01"> calls store.setMargin on onChange
// "Zapisz marże" GlowButton calls batchUpsertMarginsAction({ priceListId, margins: Object.entries(edits).map(...) })
// Show toast on success/error
// Input for a group with no existing margin defaults to "" (empty) not 0 — user must enter value
```

**`src/app/(dashboard)/price-lists/my/page.tsx`** — Server Component:

```tsx
import { requireAuth } from "@/lib/dal/auth";
import { getUserPriceList } from "@/lib/dal/price-lists";
import { auth } from "@/lib/auth"; // Auth.js v5 — get session
import { PageHeader } from "@/components/ui/page-header";
import { GlassCard } from "@/components/ui/glass-card";

export default async function MyPriceListPage() {
  const session = await auth();
  await requireAuth();
  const userId = session?.user?.id ? Number(session.user.id) : null;
  if (!userId) return <div>Brak sesji</div>;

  const priceList = await getUserPriceList(userId);

  if (!priceList) {
    return (
      <>
        <PageHeader title="Mój cennik" />
        <GlassCard>
          <div className="px-6 py-6">
            <p className="text-muted-foreground">Nie masz przypisanego cennika. Skontaktuj się z administratorem.</p>
          </div>
        </GlassCard>
      </>
    );
  }

  return (
    <>
      <PageHeader title={`Cennik: ${priceList.name}`} description={priceList.description ?? undefined} />
      <GlassCard>
        <div className="px-6 py-6">
          <table className="w-full text-sm">
            <thead>
              <tr>
                <th className="text-left py-2">Grupa produktowa</th>
                <th className="text-right py-2">Marża (%)</th>
              </tr>
            </thead>
            <tbody>
              {priceList.margins.map((m) => (
                <tr key={m.productGroupId} className="border-t border-white/10">
                  <td className="py-2">{m.productGroupName}</td>
                  <td className="py-2 text-right">{m.marginPercent.toFixed(2)}%</td>
                </tr>
              ))}
            </tbody>
          </table>
          {priceList.margins.length === 0 && (
            <p className="text-muted-foreground">Cennik nie ma jeszcze ustawionych marż.</p>
          )}
        </div>
      </GlassCard>
    </>
  );
}
```

Note: PRICE-04 says "user can view their assigned price list with correct calculated prices". For Phase 3, show margin percentages per product group. Full per-product sale price calculation is Phase 4 (quotation line items). The user view here shows the applicable margin matrix — this satisfies PRICE-04 without over-building.
