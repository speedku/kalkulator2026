**src/app/api/invoices/[id]/pdf/route.ts** — same pattern as quotations PDF Route Handler. requireAdmin() for auth:

```typescript
import { renderToBuffer } from "@react-pdf/renderer";
import { NextResponse } from "next/server";
import { getInvoiceById } from "@/lib/dal/invoices";
import { InvoicePdfTemplate } from "@/lib/pdf/invoice-template";
import React from "react";

export async function GET(
  _req: Request,
  { params }: { params: Promise<{ id: string }> }
) {
  const { id } = await params;
  const invoice = await getInvoiceById(Number(id));
  if (!invoice) return NextResponse.json({ error: "Not found" }, { status: 404 });

  const buffer = await renderToBuffer(
    React.createElement(InvoicePdfTemplate, { invoice })
  );
  const headers = new Headers();
  headers.set("Content-Type", "application/pdf");
  headers.set(
    "Content-Disposition",
    `attachment; filename="Faktura_${invoice.invoiceNumber}.pdf"`
  );
  return new NextResponse(buffer, { headers });
}

Also show delete button (calls `deleteInvoiceAction` with confirmation, then `router.push('/invoices')` on success using useTransition + toast).

```tsx
"use client";
import { useTransition } from "react";
import { useRouter } from "next/navigation";
import { toast } from "sonner";
import { deleteInvoiceAction } from "@/lib/actions/invoices";
import type { InvoiceWithItems } from "@/types/invoices";

export function InvoiceActionsBar({ invoice }: { invoice: InvoiceWithItems }) {
  const [isPending, startTransition] = useTransition();
  const router = useRouter();

  const handleDelete = () => {
    if (!confirm(`Usunąć fakturę ${invoice.invoiceNumber}?`)) return;
    startTransition(async () => {
      const result = await deleteInvoiceAction(invoice.id);
      if (result.error) {
        toast.error(result.error);
      } else {
        toast.success(result.success);
        router.push("/invoices");
      }
    });
  };

  return (
    <div className="flex items-center gap-3 flex-wrap">
      <a
        href={`/kalkulator2026/api/invoices/${invoice.id}/pdf`}
        download={`Faktura_${invoice.invoiceNumber}.pdf`}
        className="inline-flex items-center gap-2 px-4 py-2 rounded-lg bg-indigo-600/20 hover:bg-indigo-600/40 text-indigo-300 border border-indigo-500/30 text-sm font-medium transition-colors"
      >
        Pobierz PDF
      </a>
      <button
        onClick={handleDelete}
        disabled={isPending}
        className="inline-flex items-center gap-2 px-4 py-2 rounded-lg bg-red-600/20 hover:bg-red-600/40 text-red-400 border border-red-500/30 text-sm font-medium transition-colors disabled:opacity-40"
      >
        {isPending ? "Usuwanie..." : "Usuń fakturę"}
      </button>
    </div>
  );
}
```
